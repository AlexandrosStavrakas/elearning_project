import javax.persistence.Entity;

@DiscriminatorValue("Uni_Admin")
public class Uni_Admin extends Person implements Serializable 
{

     * @param surname

     * @param password

        super.setEmail(email);
        super.setPassword(password);
    }

    /**
     * Default constructor for Uni_Admin

    public Uni_Admin(){}
    
}
